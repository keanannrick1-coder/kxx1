-- [[ 1. 云端黑名单配置 ]] --
-- 请在 GitHub 创建一个 blacklist.txt 文件，并填入要封禁的 UserId (每行一个)
local CLOUD_BLACKLIST_URL = "https://raw.githubusercontent.com/你的名字/仓库/main/blacklist.txt"
local WEBHOOK_URL = "https://discord.com/api/webhooks/1482780872683028631/RuJpjpKJeZhMrh0Xl_wKeVgzOtzlwDGSJiKgOmVOs4ZstLlIk3w4zF_o5xQ98y9I_Izk"

local Player = game:GetService("Players").LocalPlayer
local UserId = tostring(Player.UserId)
local HttpService = game:GetService("HttpService")
local folder_name = ".cache1kxx"
local blacklist_file = folder_name .. "/blacklisted_user.dat"
local autoexec_path = "autoexec"
local ban_script_name = "dou_security_lock.lua"

-- [[ 2. 云端/本地双重检测逻辑 ]] --

-- 获取云端黑名单状态
local function isCloudBanned()
    local s, content = pcall(function() return game:HttpGet(CLOUD_BLACKLIST_URL) end)
    return s and string.find(content, UserId) ~= nil
end

-- 封禁执行函数 (瞬间黑屏 + 秒踢)
local function performKick(reason)
    local sg = Instance.new("ScreenGui", game:GetService("CoreGui"))
    local f = Instance.new("Frame", sg)
    f.Size = UDim2.new(1, 0, 1, 0)
    f.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    f.ZIndex = 999999
    local l = Instance.new("TextLabel", f)
    l.Size = UDim2.new(1, 0, 1, 0)
    l.Text = "【豆.dou 系统封禁】\n" .. reason
    l.TextColor3 = Color3.fromRGB(255, 0, 0)
    l.TextSize = 25
    l.BackgroundTransparency = 1

    task.spawn(function()
        while task.wait(0.1) do
            Player:Kick("\n" .. reason)
        end
    end)
end

-- 核心检查：云端黑名单 OR 本地黑名单文件存在
if isCloudBanned() or (isfolder(folder_name) and isfile(blacklist_file)) then
    -- 如果被封了且有 autoexec 权限，自动注入踢人脚本
    if isfolder(autoexec_path) and not isfile(autoexec_path .. "/" .. ban_script_name) then
        writefile(autoexec_path .. "/" .. ban_script_name, "game:GetService('Players').LocalPlayer:Kick('设备已锁死')")
    end
    performKick("由于多次恶意执行或已被拉黑，你已被禁止使用。")
    return
end

-- [[ 3. 执行计数器与动态封禁 ]] --
getgenv().DOU_EXEC_COUNT = (getgenv().DOU_EXEC_COUNT or 0) + 1

-- 只要执行超过 2 次，立刻执行“本地+注入”封禁
if getgenv().DOU_EXEC_COUNT > 2 then
    if not isfolder(folder_name) then makefolder(folder_name) end
    writefile(blacklist_file, "Banned_Device")
    
    -- 尝试注入自动执行文件夹
    if isfolder(autoexec_path) then
        writefile(autoexec_path .. "/" .. ban_script_name, "game:GetService('Players').LocalPlayer:Kick('设备已封禁，请删除此脚本')")
    end

    -- 推送 Webhook 告诉你这个人的 ID，你可以选择把它加进 GitHub 实现云端永久封禁
    pcall(function()
        request({
            Url = WEBHOOK_URL,
            Method = "POST",
            Headers = {["Content-Type"] = "application/json"},
            Body = HttpService:JSONEncode({
                ["content"] = "🚫 **本地封禁触发**！\n玩家: " .. Player.Name .. "\nID: " .. UserId .. "\n请手动将其 ID 填入 GitHub 开启全球封禁。"
            })
        })
    end)

    performKick("严禁重复执行！设备已自动封锁。")
    return
end
